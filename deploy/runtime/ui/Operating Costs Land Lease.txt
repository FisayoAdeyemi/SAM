equations{ 'om_land_lease'} = define(){
	if ( ${ui_land_area_units} == 1 )
	{
		for ( i=0; i<#${ui_land_lease}; i++)
		{
			x[i] = ${ui_land_lease}[i] * 2.47105;
		}
	}
	else
	{
		x = ${ui_land_lease};
	}
	return x;
};

equations{ 'om_land_area' } = define() {
	return ${ui_collector_area} * ${ui_land_area_multiplier};
};

equations{ 'ui_collector_area' } = define() 
{
	if ( technology() == 'PVWatts' )
	{ 
		// land area estmate https://www.nrel.gov/analysis/tech-size.html
		eff = ?${module_type}[0.17,0.201,0.156];
		return ${system_capacity}*eff / ${gcr} * 0.0002471; // area (m2) = capacity (kW) / 1 (kW/m2)
	}
	elseif ( technology() == 'Flat Plate PV' )
	{
		return ${total_land_area};
	}
	else
	{
		return 0;
	}
};

equations{ 'ui_land_area_ha' } = define()
{
	return ${om_land_area} * 0.404686;
};

function enable_lease_escalation() 
{
	enable( 'om_land_lease_escal',#${ui_land_lease}==1);
	units=['$/acre','$/ha'];
	property('ui_land_lease','Label',units[${ui_land_area_units}]);
	property( 'ui_land_lease','FixedLength',${analysis_period});
}
