on_load{'Financial CS Subscribers'} = define()
{
	property('subscriber_share_anchor','FixedLength',${analysis_period});
	property('subscriber_share_general','FixedLength',${analysis_period});
	property('subscriber_share_subsidized','FixedLength',${analysis_period});
	enable_growth_rate();
	subscriber_share_message();
};

on_change{'subscriber_share_anchor'} = define()
{
	enable('subscriber_growth_rate_anchor', #${subscriber_share_anchor} == 1 );
	subscriber_share_message();
};

on_change{'subscriber_share_general'} = define()
{
	enable('subscriber_growth_rate_general', #${subscriber_share_general} == 1 );
	subscriber_share_message();
};

on_change{'subscriber_share_subsidized'} = define()
{
	enable('subscriber_growth_rate_subsidized', #${subscriber_share_subsidized} == 1 );
	subscriber_share_message();
};

function enable_growth_rate() 
{
	enable('subscriber_growth_rate_anchor', #${subscriber_share_anchor} == 1 ); 
	enable('subscriber_growth_rate_general', #${subscriber_share_general} == 1 ); 
	enable('subscriber_growth_rate_subsidized', #${subscriber_share_subsidized} == 1 ); 
}

function subscriber_share_message()
{
	total = ${subscriber_share_anchor}[0]
	+ ${subscriber_share_general}[0]
	+ ${subscriber_share_subsidized}[0];
	if ( total > 100 )
	{
		msg = 'The project is oversubscribed at ' + total + '% in Year 1! Reduce the anchor, general, or subsidized subscriber subscription rate so that the total is 100% or less.'; 
		color = 'red';
	}
	else
	{
		msg = 'The project is ' + total + '% subscribed in the first year. See the unsubscribed rate in the results after running a simulation.';
		color = 'black';
	}
	property('msg_subscriber_share','Caption',msg);
	property('msg_subscriber_share','TextColour',color);}
