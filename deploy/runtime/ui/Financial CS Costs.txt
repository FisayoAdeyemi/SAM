on_load{'Financial CS Costs'} = define()
{
	enable_cost_escalation_rate();
	property( 'subscriber_cost_recurring_fixed','FixedLength',${analysis_period});
	property( 'subscriber_cost_recurring_capacity','FixedLength',${analysis_period});
	property( 'subscriber_cost_recurring_generation','FixedLength',${analysis_period});

};

on_change{'subscriber_cost_recurring_fixed'} = define()
{
	enable_cost_escalation_rate();
};

on_change{'subscriber_cost_recurring_capacity'} = define()
{
	enable_cost_escalation_rate();
};

on_change{'subscriber_cost_recurring_generation'} = define()
{
	enable_cost_escalation_rate();
};

function enable_cost_escalation_rate() 
{
	enable('subscriber_cost_recurring_fixed_escal', #${subscriber_cost_recurring_fixed} == 1 ); 
	enable('subscriber_cost_recurring_capacity_escal', #${subscriber_cost_recurring_capacity} == 1 ); 
	enable('subscriber_cost_recurring_generation_escal', #${subscriber_cost_recurring_generation} == 1 ); 
}
