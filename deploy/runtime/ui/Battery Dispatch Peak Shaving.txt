on_change{'peak_shaving_batt_dispatch_choice'} = define() {
	peak_shaving_choice();
};

function peak_shaving_choice() {
	choice = value('peak_shaving_batt_dispatch_choice');
	if (choice == 0) {
		value('batt_dispatch_choice',0);
	}
	else if (choice == 1) {
		value('batt_dispatch_choice', 1);
	}
}
